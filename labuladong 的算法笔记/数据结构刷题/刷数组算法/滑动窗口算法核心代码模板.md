class Solution:  
    def lengthOfLongestSubstring(self, s: str) -> int:  
        l, r = 0, 0  
        length = 0  
        window = dict()  
  
        while r < len(s):  
            c = s[r]  
            r += 1  
            window[c] = window.get(c, 0) + 1  
  
            while window[c] > 1:  
                d = s[l]  
                l += 1  
                window[d] -= 1  
  
            length = max(length, r - l)  
  
        return length
