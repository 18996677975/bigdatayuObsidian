```python
class ListNode:

        self.val = val  
        self.next = next

    def removeNthFromEnd(self, head: Optional[ListNode], n: int) -> Optional[ListNode]:  
        dummy = ListNode(-1)  
        dummy.next = head  
        fast, slow = dummy, dummy  
  
        for i in range(n + 1):  
            fast = fast.next  
  
        while fast:

        slow.next = slow.next.next  
  
        return dummy.next
```

```python
class Solution:  
    def detectCycle(self, head: Optional[ListNode]) -> Optional[ListNode]:  
        fast, slow = head, head  
        while fast and fast.next:  
            fast = fast.next.next  
            slow = slow.next  
            if fast == slow:  
                break  
  
        if not fast or not fast.next:  
            return None  
  
        fast = head  
  
        while fast != slow:  
            fast = fast.next  
            slow = slow.next  
  
        return slow
```

```python
class ListNode:  
    def __init__(self, x):  
        self.val = x  
        self.next = None

    def getIntersectionNode(self, headA: ListNode, headB: ListNode) -> Optional[ListNode]:  
        p1, p2 = headA, headB  
  
        while p1 != p2:  
            if p1:  
                p1 = p1.next
```
